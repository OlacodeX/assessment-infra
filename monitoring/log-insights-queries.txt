# Recent backend logs

fields @timestamp, @message
| sort @timestamp desc
| limit 50


# Find backend errors

fields @timestamp, @message
| filter @message like /ERROR/
| sort @timestamp desc
| limit 20


# Find slow API requests

fields @timestamp, @message
| filter @message like /slow/
| sort @timestamp desc
| limit 20


# ALB related failures

fields @timestamp, @message
| filter @message like /5XX/
| sort @timestamp desc
| limit 20
